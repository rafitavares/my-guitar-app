    <h2>🎸 Chord Finder</h2>

    

    <input type="text" id="searchInput" placeholder="Search for a song (e.g., Wonderwall)">

    <button onclick="searchSong()">Search</button>


    <div id="resultArea" style="display: none;">

        <h3 id="songTitle"></h3>

        <div id="songChords" class="chord-box"></div>

        <button onclick="saveToPlaylist()">❤️ Save to Playlist</button>

    </div>


    <hr>


    <h3>My Playlist</h3>

    <div id="playlistArea">

        </div>

<script>

    // 1. Our "Mock" Database of Songs

    const songDatabase = [

        { title: "Wonderwall", chords: "[Em] Today is [G] gonna be the day\nThat they're [D] gonna throw it back to [A7sus4] you" },

        { title: "Let It Be", chords: "When I [C] find myself in [G] times of trouble\n[Am] Mother Mary [F] comes to me" }

    ];


    let currentSong = null; // Keeps track of what is on the screen


    // 2. Search Function

    function searchSong() {

        const query = document.getElementById("searchInput").value.toLowerCase();

        const result = songDatabase.find(song => song.title.toLowerCase().includes(query));


        if (result) {

            currentSong = result;

            document.getElementById("songTitle").innerText = result.title;

            document.getElementById("songChords").innerText = result.chords;

            document.getElementById("resultArea").style.display = "block";

        } else {

            alert("Song not found! Try 'Wonderwall' or 'Let It Be'.");

            document.getElementById("resultArea").style.display = "none";

        }

    }


    // 3. Save to LocalStorage Function

    function saveToPlaylist() {

        if (!currentSong) return;


        // Get existing playlist from the phone's memory, or start an empty array

        let playlist = JSON.parse(localStorage.getItem("myPlaylist")) || [];


        // Check if song is already saved

        const alreadySaved = playlist.some(song => song.title === currentSong.title);

        

        if (!alreadySaved) {

            playlist.push(currentSong);

            localStorage.setItem("myPlaylist", JSON.stringify(playlist));

            alert(currentSong.title + " saved to your playlist!");

            displayPlaylist(); // Update the screen

        } else {

            alert("This song is already in your playlist.");

        }

    }


    // 4. Display Playlist Function

    function displayPlaylist() {

        const playlistArea = document.getElementById("playlistArea");

        playlistArea.innerHTML = ""; // Clear it out first

        

        let playlist = JSON.parse(localStorage.getItem("myPlaylist")) || [];


        if (playlist.length === 0) {

            playlistArea.innerHTML = "<p style='color: gray; font-size: 14px;'>No songs saved yet.</p>";

            return;

        }


        // Loop through saved songs and put them on the screen

        playlist.forEach(song => {

            const div = document.createElement("div");

            div.className = "song-item";

            div.innerHTML = `<strong>${song.title}</strong>`;

            playlistArea.appendChild(div);

        });

    }


    // Run this when the app loads to show already saved songs

    displayPlaylist();

</script>
